CREATE TABLE ticket_by_place_from_place_to_takeoff_day (
    id uuid, 
    country_from text, 
    city_from text, 
    country_to text, 
    city_to text, 
    takeoff_year int,
    takeoff_month int,
    takeoff_day int,
    takeoff_date timestamp, 
    arrive_date timestamp, 
    company text,
    transit_places text, 
    price int, 
    PRIMARY KEY((country_from, city_from, country_to, city_to, takeoff_year, takeoff_month, takeoff_day), id));
    
CREATE TABLE ticket_by_id (
        id uuid, 
        country_from text, 
        city_from text, 
        country_to text, 
        city_to text, 
        takeoff_year int,
        takeoff_month int,
        takeoff_day int,
        takeoff_date timestamp, 
        arrive_date timestamp, 
        transit_places text, 
        company text,
        price int, 
        PRIMARY KEY (id));
        
CREATE TABLE warrants_by_passenger_idno(
    id uuid,
    passenger_idno text,
    passenger_passport_id text,
    ticket_id uuid,
    ticket_back_id uuid,
    is_paid boolean,
    PRIMARY KEY ((passenger_idno), id)
);

CREATE TABLE warrant_by_passenger_idno_and_ticket_id(
    id uuid,
    passenger_idno text,
    passenger_passport_id text,
    ticket_id uuid,
    ticket_back_id uuid,
    is_paid boolean,
    PRIMARY KEY ((passenger_idno, ticket_id), id)
);
